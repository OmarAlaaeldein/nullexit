iptables -A FORWARD -i tailscale0 -o tun0 -j ACCEPT
iptables -A FORWARD -i tun0 -o tailscale0 -j ACCEPT
iptables -A INPUT -i tailscale0 -j ACCEPT
iptables -A INPUT -i eth0 -p udp --dport 41641 -j ACCEPT
iptables -t nat -A POSTROUTING -o tun0 -j MASQUERADE
# Clamp MSS to 1120. With Tailscale overhead on top of WARP (each WireGuard encapsulation adds ~60 bytes),
# the safe MSS for double-tunneled traffic is 1120 to prevent strict-MSS endpoints from stalling.
iptables -t mangle -A POSTROUTING -p tcp --tcp-flags SYN,RST SYN -j TCPMSS --set-mss 1120

iptables -t nat -A PREROUTING -i tailscale0 -p udp --dport 53 -j REDIRECT --to-port 5335
iptables -t nat -A PREROUTING -i tailscale0 -p tcp --dport 53 -j REDIRECT --to-port 5335

# Also redirect DNS from Docker bridge interface so the host can reach AdGuard via localhost:53
iptables -t nat -A PREROUTING -i eth0 -p udp --dport 53 -j REDIRECT --to-port 5335
iptables -t nat -A PREROUTING -i eth0 -p tcp --dport 53 -j REDIRECT --to-port 5335

# Block IPv6 exit-node traffic to prevent leakage (Gluetun/WARP is IPv4-only)
ip6tables -A FORWARD -i tailscale0 -j DROP

# Block DNS-over-TLS (Port 853) to force Android Private DNS to fallback to Port 53
iptables -A FORWARD -i tailscale0 -p tcp --dport 853 -j REJECT --reject-with tcp-reset
iptables -A FORWARD -i tailscale0 -p udp --dport 853 -j REJECT
